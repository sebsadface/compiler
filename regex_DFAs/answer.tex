            (i) \textbf{Can be generated:} $\mathrm{axy}$, $\mathrm{aaxy}$ \;\;\;\textbf{Cannot be generated:} $\mathrm{ayx}$, $\mathrm{axay}$ \\
            (ii) \textbf{English description:} Any number of repetitions of 'a' or 'xy' in any order, including the empty string.
        \end{answer}
    \end{part}

    \begin{part}
        \begin{answer}
            (i) \textbf{Can be generated:} $\mathrm{bozo}$, $\mathrm{bozozo}$ \;\;\;\textbf{Cannot be generated:} $\mathrm{bzoo}$, $\mathrm{boozzo}$ \\
            (ii) \textbf{English description:} Strings start with a 'b', followed by one or more 'oz', and end with an 'o'.
        \end{answer}
    \end{part}

    \begin{part}
        \begin{answer}
            (i) \textbf{Can be generated:} $\mathrm{01}$, $\mathrm{0101}$ \;\;\;\textbf{Cannot be generated:} $\mathrm{10}$, $\mathrm{0110}$ \\
            (ii) \textbf{English description:}  Any number of repetitions of '01' or '1' in any order, including the empty string.
        \end{answer}
    \end{part}
