            \begin{answer}
                \textbf{Set of variables that are live at the beginning of each basic block:}
                \begin{itemize}
                    \item \textbf{B1: } \{n\}
                    \item  \textbf{B2: } \{n, v, m\}
                    \item \textbf{B3: } \{n, v, m\}
                    \item \textbf{B4: } \{r, n, v, s, m\}
                    \item \textbf{B5: } \{n, v, m\}
                    \item \textbf{B6: } \{r, n, v, s, m\}
                    \item \textbf{B7: } \{r, n, v, s\}
                    \item \textbf{B8: } \{r, n, v, s, m\}
                    \item \textbf{B9: } \{m\}
                \end{itemize}
            \end{answer}
    \end{part}

    \begin{part}
        \begin{answer}
            \textbf{Uninitialized variable:} n\\
            From the live variable analysis, we can see that the variable n is live at the beginning of block 1, but block 1 is the first block in the program (does not have any predecessors). Therefore, the variable n used in block 2 and 4 is uninitialized.
            All other variables are initialized before they are used.
        \end{answer}
    \end{part}
